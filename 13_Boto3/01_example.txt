import boto3

client = boto3.client('s3') # you can change to resource as well like ec2, iam etc.

# create a new bucket
response = client.create_bucket(

    Bucket='my-first-bucket-1234567890'
)

# get bucket ACL
response = client.get_bucket_acl(
    Bucket='my-first-bucket-1234567890'
)

print(response)
